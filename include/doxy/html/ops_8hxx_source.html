187  constexpr bool is_deref_instruction(opcode op) noexcept
188  {
189  return (opcode::mov_eax_ptr_eax <= op && op <= opcode::mov_esp_ptr_esp) ||
190  (opcode::mov_eax_ptr <= op && op <= opcode::mov_edi_ptr);
