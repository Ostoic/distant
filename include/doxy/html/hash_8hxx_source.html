20  template <std::size_t S>
21  constexpr std::size_t hash_bytes(const std::array<unsigned char, S>& bytes) noexcept
22  {
23  // This is just an implementation of the Fowler–Noll–Vo (FNV) hash,
24  // using Visual Studio's primes and basis taken from _Hash_Bytes.
25 #if defined(_WIN64)
26  static_assert(sizeof(std::size_t) == 8, "This code is for 64-bit std::size_t.");
27  const std::size_t offset_basis = 14695981039346656037ULL;
28  const std::size_t prime = 1099511628211ULL;
29 #else
30  static_assert(sizeof(std::size_t) == 4, "This code is for 32-bit std::size_t.");
31  const std::size_t offset_basis = 2166136261U;
32  const std::size_t prime = 16777619U;
33 #endif
34 
35  std::size_t result = offset_basis;
36  for (std::size_t i = 0; i < S; ++i)
37  {
38  result ^= static_cast<std::size_t>(bytes[i]);
39  result *= prime;
40  }
41 
42  return result;
43  }
44  }
