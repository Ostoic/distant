51  using index = std::size_t;

53  using op_length = std::size_t;

55 
56  // The opcode can be represented by a byte
57  return static_assembler<sizeof...(Bytes) + sizeof(opcode), 1> {
58  std::array<distant::byte, sizeof...(Bytes) + opcode_length(op)> {
59  get_byte<0>(static_cast<under>(op)),
60  get_byte<1>(static_cast<under>(op)),
61  static_cast<distant::byte>(bytes)...
62  },
63 
64  {std::make_tuple(0, opcode_length(op), sizeof...(Bytes) + 2)}
