93  {
94  BOOST_ASSERT_MSG(this->valid(), "[process_base::is_being_debugged] invalid process handle");
95 
96  // If we are considering the current process, use IsDebuggerPresent.
97  if (this->id_ == GetCurrentProcessId())
98  return IsDebuggerPresent();
99 
100  // Otherwise use CheckRemoteDebuggerPresent.
101  boost::winapi::BOOL_ result = false;
102  boost::winapi::CheckRemoteDebuggerPresent(this->handle_.native_handle(), &result);
103  return result;

106  inline bool process_base::is_zombie() const
107  {
108  /* Use of GetProcessVersion:
109  ** I ran into a bug where OpenProcess would return a "valid" handle that did not refer to any process.
110  ** It was apparent that no such process existed because under task manager/procmon, no such id was
111  ** listed. I noticed that GetProcessVersion would perform some sort of integrity check on the process
112  ** id, which seemed to be consistent with processes on the manager. Indeed, the process handles I was
113  ** getting coincided with those from using the standard ToolHelp functions.
114  ** Note: I didn't realize zombie processes exist on Windows in this form.
115  */
116 
117 #pragma warning(push)
118 #pragma warning(disable:4267)
119  return handle_ != nullptr && GetProcessVersion(this->id_) == 0;
120 #pragma warning(pop)
121  }
