84  template <access_rights::token A, typename K>
85  bool access_token<A, K>::set_privilege(const security::privilege& p,
86  security::privilege::attributes attribute) noexcept
87  {
88  if (!p) return false;
89 
90  boost::winapi::TOKEN_PRIVILEGES_ temp = p;
91  temp.Privileges[0].Attributes = static_cast<boost::winapi::DWORD_>(attribute);
92 
93  if (boost::winapi::adjust_token_privilege(this->handle().native_handle(), false, &temp, sizeof(temp), nullptr,
94  nullptr))
95  {
96  // If the privilege was set return success.
97  if (last_error().value() != boost::winapi::ERROR_NOT_ALL_ASSIGNED_)
98  return true;
99  }
100 
101  // Otherwise an error has occured.
102  return false;
103  }
